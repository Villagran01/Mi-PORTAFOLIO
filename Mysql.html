1. Creación de Base y Tablas (DDL & Inserts)


        CREATE DATABASE tienda;
        USE tienda;

        DROP TABLE IF EXISTS clientes;
        DROP TABLE IF EXISTS pedidos;

        -- Crear tabla CLIENTES
        CREATE TABLE clientes (
            id_cliente INT AUTO_INCREMENT PRIMARY KEY,
            nombre VARCHAR(100) NOT NULL,
            direccion VARCHAR(150),
            email VARCHAR(100) UNIQUE,
            telefono VARCHAR(20),
            fecha_registro DATE
        );

        -- Crear tabla PEDIDOS
        CREATE TABLE pedidos (
            id_pedido INT AUTO_INCREMENT PRIMARY KEY,
            id_cliente INT NOT NULL,
            producto VARCHAR(100) NOT NULL,
            cantidad INT NOT NULL,
            fecha DATE,
            estado VARCHAR(50),
            FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente)
        );

2. Ejemplos de Consultas (DML & SELECT)


        -- 2.1 Obtener todos los pedidos de 'Juan Pérez'
        SELECT p.id_pedido, p.producto, p.cantidad, p.fecha, p.estado
        FROM pedidos p
        JOIN clientes c ON p.id_cliente = c.id_cliente
        WHERE c.nombre = 'Juan Pérez';

        -- 2.2 Nombres de clientes que han realizado pedidos
        SELECT DISTINCT c.nombre
        FROM clientes c
        JOIN pedidos p ON c.id_cliente = p.id_cliente;

        -- 2.3 Total de productos pedidos por cliente
        SELECT c.nombre, SUM(p.cantidad) AS total_productos
        FROM clientes c
        JOIN pedidos p ON c.id_cliente = p.id_cliente
        GROUP BY c.nombre;

        -- 2.4 Pedidos con estado 'Pendiente'
        SELECT p.id_pedido, c.nombre, p.producto, p.cantidad, p.fecha
        FROM pedidos p
        JOIN clientes c ON p.id_cliente = c.id_cliente
        WHERE p.estado = 'Pendiente';

        -- 2.5 Todos los pedidos ordenados por fecha descendente
        SELECT p.id_pedido, c.nombre, p.producto, p.fecha, p.estado
        FROM pedidos p
        JOIN clientes c ON p.id_cliente = c.id_cliente
        ORDER BY p.fecha DESC;

        -- 2.6 Número total de pedidos por estado
        SELECT estado, COUNT(*) AS total_pedidos
        FROM pedidos
        GROUP BY estado;
